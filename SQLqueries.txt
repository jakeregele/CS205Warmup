ranking song SONG_NAME:

artist song SONG_NAME:

	SELECT name FROM top_artists WHERE id = artist;

genre song SONG_NAME:
	SELECT tblTop50Songs.nameOfSong, tblArtists.genre FROM tblTop50Songs INNER JOIN tblArtists ON tblTop50Songs.artist = tblArtists.id WHERE tblTop50Songs.nameofSong = ?

genre artist ARTIST_NAME:
	SELECT genre from tblArtists wHERE name = ?

topRankedSong artist ARTIST_NAME:
	SELECT tblArtists.topRankedSong, tblTop50Songs.nameOfSong FROM tblArtists
	INNER JOIN tblTop50Songs ON tblArtists.topRankedSong = tblTop50Songs.id
	WHERE tblArtists.topRankedSong = ?

ranking artist ARTIST_NAME:
	SELECT top_artists.topRankedSong, top_songs.ranking FROM top_artists
	INNER JOIN top_songs ON tblArtists.topRankedSong = tblTop50Songs.id
	WHERE tblArtists.topRankedSong = ?
